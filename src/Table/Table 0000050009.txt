        IF PurchRcptLineBuffer.FINDFIRST THEN BEGIN
          IF "Receipt No."='' THEN
          BEGIN
            LPurchRcptLine.GET(PurchRcptLineBuffer."Document No.",PurchRcptLineBuffer."Line No.")
          END
          ELSE
          BEGIN
            LPurchRcptLine.GET("Receipt No.","Receipt Line No.");

            QtyGAP := Quantity - LPurchRcptLine."Qty. Rcd. Not Invoiced";
            IF QtyGAP >= PurchRcptLineBuffer."Qty. Rcd. Not Invoiced" THEN
              QtyToInv := PurchRcptLineBuffer."Qty. Rcd. Not Invoiced"
            ELSE
              QtyToInv := QtyGAP;

            "Direct Unit Cost" := Amount / Quantity;
            InitNewGIMPurchLine(Rec,NewGimPurchLine,QtyToInv,PurchRcptLineBuffer);
            Quantity := Quantity - QtyToInv;
            Amount := Quantity * "Direct Unit Cost";
          END;

          IsError := FALSE;
          "Error log" := '';
          "Receipt No." := LPurchRcptLine."Document No.";
          Type := LPurchRcptLine.Type;
          "No." := LPurchRcptLine."No.";
          "Receipt Line No." := LPurchRcptLine."Line No.";

          MODIFY(TRUE);

        END;
      END;
    END;

    PROCEDURE InitNewGIMPurchLine@1000000004(VAR _Rec@1000000002 : Record 50009;VAR NewGimPurchLine@1000000003 : Record 50009;_QtyGAP@1000000001 : Decimal;_PurchRcptLine@1000000000 : Record 121);
    BEGIN
      NewGimPurchLine.INIT;
      NewGimPurchLine := _Rec;
      NewGimPurchLine."Line No." := 0;
      NewGimPurchLine."Receipt No." := _PurchRcptLine."Document No.";
      NewGimPurchLine."Receipt Line No." := _PurchRcptLine."Line No.";
      NewGimPurchLine.Type := _PurchRcptLine.Type;
      NewGimPurchLine."No." := _PurchRcptLine."No.";
      NewGimPurchLine.Quantity := _QtyGAP;
      NewGimPurchLine.Amount := NewGimPurchLine.Quantity * NewGimPurchLine."Direct Unit Cost";
      NewGimPurchLine."VAT Amount" := 0;
      NewGimPurchLine."Allowance charge" := 0;
      NewGimPurchLine."VAT Allowance charge" := 0;
      NewGimPurchLine.IsError := FALSE;
      NewGimPurchLine."Error log" := '';
      NewGimPurchLine.INSERT(TRUE);
    END;

    BEGIN
    {
      AC1.00 Ajout Champ "Order Reference Customer"
    }
    END.
