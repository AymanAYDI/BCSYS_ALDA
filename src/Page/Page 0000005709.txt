    PROCEDURE SetPurchHeader@1(VAR PurchHeader2@1000 : Record 38);

      PurchHeader.GET(PurchHeader2."Document Type",PurchHeader2."No.");

    LOCAL PROCEDURE IsFirstDocLine@2() : Boolean;
    VAR
      PurchRcptLine@1000 : Record 121;
    BEGIN
      TempPurchRcptLine.RESET;
      TempPurchRcptLine.COPYFILTERS(Rec);
      TempPurchRcptLine.SETRANGE("Document No.","Document No.");
      IF NOT TempPurchRcptLine.FINDFIRST THEN BEGIN
        PurchRcptLine.COPYFILTERS(Rec);
        PurchRcptLine.SETRANGE("Document No.","Document No.");
        PurchRcptLine.SETFILTER("Qty. Rcd. Not Invoiced",'<>0');
        //AB6.JME.001--<<
        //IF PurchRcptLine.FINDFIRST THEN BEGIN
        IF PurchRcptLine.FINDFIRST THEN ;
        //AB6.JME.001-->>
          TempPurchRcptLine := PurchRcptLine;
          TempPurchRcptLine.INSERT;
        //END;
      END;
      IF "Line No." = TempPurchRcptLine."Line No." THEN
        EXIT(TRUE);
    END;

    LOCAL PROCEDURE CreateLines@19031339();
    BEGIN
      CurrPage.SETSELECTIONFILTER(Rec);

      GetReceipts.CreateInvLines(Rec);
    END;

    LOCAL PROCEDURE DocumentNoOnFormat@19001080();
    BEGIN
      //IF NOT IsFirstDocLine THEN         //Code 2015
      //  "Document No.HideValue" := TRUE; //Code 2015
      //AB6.S35
      myPurchRcptLine.RESET;
      IF IsFirstDocLine THEN BEGIN
        "Document No.Emphasize" := TRUE;
         MontantFacture:=0;
         myPurchRcptLine.COPYFILTERS(Rec);
         myPurchRcptLine.SETFILTER("Document No.",Rec."Document No.");

         REPEAT
            MontantFacture+=myPurchRcptLine.Quantity*myPurchRcptLine."Unit Cost";
         UNTIL myPurchRcptLine.NEXT = 0;

      END ELSE BEGIN
        "Document No.HideValue" := TRUE;
        MontantFacture+=MontantLigne;
      END;
