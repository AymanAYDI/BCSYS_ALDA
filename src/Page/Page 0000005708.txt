    PROCEDURE SetSalesHeader@1(VAR SalesHeader2@1000 : Record 36);

    LOCAL PROCEDURE IsFirstDocLine@2() : Boolean;
    VAR
      SalesShptLine@1000 : Record 111;
    BEGIN
      TempSalesShptLine.RESET;
      TempSalesShptLine.COPYFILTERS(Rec);
      TempSalesShptLine.SETRANGE("Document No.","Document No.");
      IF NOT TempSalesShptLine.FINDFIRST THEN BEGIN
        SalesShptLine.COPYFILTERS(Rec);
        SalesShptLine.SETRANGE("Document No.","Document No.");
        SalesShptLine.SETFILTER("Qty. Shipped Not Invoiced",'<>0');
        //AB6.JME.001 --<<
        IF SalesShptLine.FINDFIRST THEN BEGIN
        //IF SalesShptLine.FINDFIRST THEN;
        //AB6.JME.001 --<<
          TempSalesShptLine := SalesShptLine;
          TempSalesShptLine.INSERT;
        END;
      END;
      IF "Line No." = TempSalesShptLine."Line No." THEN
        EXIT(TRUE);
    END;

    PROCEDURE GetUnitPriceFromOrder@1180250002();
    VAR
      SalesLine@1180250001 : Record 37;
    BEGIN
      // ALDA 031214D.Begin
      IF SalesLine.GET(SalesLine."Document Type"::Order,"Order No.","Order Line No.") THEN BEGIN
        UnitPrice := SalesLine."Unit Price";
        LineDiscountAmt := SalesLine."Line Discount Amount";
        Amt := (UnitPrice * Quantity) - LineDiscountAmt;
      END ELSE BEGIN
        UnitPrice := 0;
        LineDiscountAmt := 0;
        Amt := 0;
      END;
